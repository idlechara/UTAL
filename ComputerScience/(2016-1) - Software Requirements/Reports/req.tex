    \reqrationale{Se busca producir una aplicación portátil, y de rápido acceso.}

    \reqfitcriterion{La aplicación debe ejecutarse en teléfonos inteligentes con Sistema Operativo Android.}

\begin{requirement}{9}{---}
    \reqdescription{ La aplicación no debe requerir internet para su uso.}
    \reqrationale{Se busca una aplicación móvil educativa, para que alumnos de enseñanza media puedan estudiar en cualquier lugar, y en algunos de ellos no hay conexión estable a internet.}
    \reqsource{Renzo Angles}
    \reqfitcriterion{La aplicación se ejecuta sin necesidad de estar conectado a internet.}
    \reqsatisfaction{}{}
    \reqdepencies{}{}
    \reqdocuments{}
    \reqhistory{}

\end{requirement}


\begin{requirement}{9}{---}

    \reqfitcriterion{Al presionar un botón, se debe realizar la transición de una ventana a otra.}
