\section{Descripción}
\begin{frame}
\frametitle{Situación a analizar}
Arnold Bosch está a cargo del departamento de ``Control de entradas'' en una empresa de textiles alemana la cual actualmente está al borde de una crisis debido a problemas en el suministro. Estudios llevados a cabo en la empresa están actualmente apuntando sus dedos al departamento en el que A. Bosch está a cargo.
\end{frame}


\section{Principales falencias}
\begin{frame}
\frametitle{Situación a analizar}
\begin{itemize}
	\item Existencia de un líder informal y posterior pérdida de este.
	\item Falta de proactividad de parte de Bosch para resolver problemas y poca presencia en los procesos.
	\item Baja identificación con la tarea de parte del personal
\end{itemize}
\end{frame}



\section{Repercusión de fallos}
\begin{frame}
\frametitle{Repercusión de fallos en el departamento de ``Control de entradas'' en la producción}
\begin{itemize}
	\item Retraso en envios
	\item Fallas mecánicas en maquinaria producto de la baja calidad del material
	\item Desmotivación del personal de producción (debido a la razón anterior)
\end{itemize}
\end{frame}


\section{Medidas a tomar}
\begin{frame}
\frametitle{En general}
\begin{itemize}
	\item Reemplazo del líder informal
	\item Implementar políticas de control
	\item Analizar el estado de los trabajadores
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Delegación}
\begin{itemize}
	\item Analizar permanencia del cargo
	\item Analizar nivel de identificación con la tarea
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Implementación}
\begin{itemize}
	\item Solucionar problema de falta de motivación
	\item Falta de mecanismos de control
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Control}
\begin{itemize}
	\item Generar espacio para presentar inquetudes
	\item Establecer un control durante el proceso
	\item Evaluaciones de desempeño y estado
\end{itemize}
\end{frame}
