	\reqdescription{Un proyecto puede lanzar una instancia temporal de ejecución de estar disponible.}
	\reqrationale{Algunos códigos fuente o paquetes ejecutables pueden ser desplegados desde dentro de la apliación. En estos casos, si el administrador habilita la opción de ejecución temporal, debe de haber
	una opción disponible en la vista del proyecto.}
	\reqsource{Rodrigo Bustamante}
	\reqfitcriterion{Administrador puede indicar que un proyecto puede ser ejecutado directamente desde

\end{requirement}

\begin{requirement}{functional}{}
	\reqdescription{Las instancias temporales de ejecución son únicas y por defecto están globalmente disponibles 
	para los usuarios con los permisos adecuados.}
	\reqrationale{Algunos códigos fuente o paquetes ejecutables pueden ser desplegados desde dentro de la apliación. 
	En estos casos, si el administrador habilita la opción de ejecución temporal, debe de haber
	una opción disponible en la vista del proyecto. Sin embargo, no tiene sentido ejecutar múltiples 
	instancias de una misma aplicación, esencialmente por un tema de recursos de la organización. Si bien 
	puede existir el caso de un usuario que requiera un ambiente más controlado, como no es el caso general 
	la opción por defecto es que sea una instancia de la ejecución global.}
	\reqsource{I.O.R.S.H.}
	\reqfitcriterion{Cuando más de un usuario lanza la misma aplicación virtual esta por defecto lo redirige a la
	instancia global.}
	\reqsatisfaction{2}{3}
	\reqdepencies{\reffr{RF:P:REVIEW_PUBLISH}, \reffr{RF:U:PRIVILEGIO}, \reffr{RF:P:VIRTUAL_ENVIRONMENT_CONFIG}}{}
	\reqdocuments{}
	\reqhistory{}
	\label{RF:P:VIRTUAL_ENVIRONMENT_GLOBAL}
\end{requirement}
