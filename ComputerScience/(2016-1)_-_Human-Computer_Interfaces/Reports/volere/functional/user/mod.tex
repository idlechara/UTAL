\subsection{Administración moderacion}
\begin{requirement}{functional}{PUC\_10,PUC\_11}
	\reqdescription{Los moderadores pueden agregar, eliminar y modificar filtros de lenguaje.}
	\reqrationale{Los moderadores son encargados de velar por la correcta conducta en los comentarios y
	 contenido. Por tanto, ellos pueden implementar filtros de lenguaje los cuales asisten a la 
	 automatización de la revisión.}
	\reqsource{Rodrigo Bustamante}
	\reqfitcriterion{El moderador puede modificar y crear filtros de lenguaje desde la vista de moderación.}
	\reqsatisfaction{2}{2}
	\reqdepencies{}{}
	\reqdocuments{Entrevista 4}
	\reqhistory{}
	\label{RF:U:MOD_FILTER}
\end{requirement}

\begin{requirement}{functional}{}
	\reqdescription{Los moderadores no pueden agregar proyectos.}
	\reqrationale{Los moderadores son encargados de velar por la correcta conducta en los comentarios y
	 contenido. Por tanto, ellos no necesitan subir proyectos.}
	\reqsource{Rodrigo Bustamante}
	\reqfitcriterion{El moderador tiene deshabilitada la opción de subir proyectos.}
	\reqsatisfaction{2}{2}
	\reqdepencies{}{}
	\reqdocuments{Entrevista 6}
	\reqhistory{}
	\label{RF:U:MOD_NOT_UPLOAD}
\end{requirement}
