\section{Resúmen ejecutivo}
  \begin{frame}
    \frametitle{Resúmen ejecutivo}
    \begin{itemize}
      \item WakeUP! presenta una solución a regular problemas del sueño por medio de
      \emph{fototerapia} sin necesidad de contar con un ambiente simulado.
      \item Proyecto rentable, pero arriesgado. Potencial mercado de 7.8\% de la población chilena.
      \item \emph{Tiempo de retorno:} 3 años, 5.270.838 CLP
      \item \emph{VAN:} 22.397.617 CLP 
      \item \emph{TIR:}  53\%
    \end{itemize}
  \end{frame}


\section{Descripción de la empresa}
  \begin{frame}
    \frametitle{Descripción de la empresa}
    \begin{itemize}
      \item Housetic
      \item Dos socios: Alexis Angulo y Erik Regla
      \item Nuestro producto: WakeUP!
      \item Permite la apertura de cortinas basada en información del ambiente,
      para así reducir o mitigar los problemas producidos por desordenes del
      ritmo del sueño.
    \end{itemize}
  \end{frame}


\section{Análisis de mercado}
  \begin{frame}
    \frametitle{Problema / Necesidad}
    \begin{itemize}
      \item 60\% de la población chilena padece de problemas del sueño
      \item No es un asunto abordado típicamente, tiende a ser ignorado.
      \item Problema más común: \emph{Sindrome de fase retrasada del sueño}
      \item \emph{Fototerápia} artificial es utilizada como solución.
    \end{itemize}
  \end{frame}
\subsection{Nuestros clientes}
  \begin{frame}
    \frametitle{Clientes}
    \begin{itemize}
      \item{Personas con problemas del sueño, entre 25 a 52 años pertenecientes a los últimos 5 
      deciles socioeconómicos.}
      \item{Imobiliarias y contructoras que deseen implementar nuestra solución en sus proyectos 
      de construcción.}
    \end{itemize}
  \end{frame}

  \subsection{Análisis de mercado}
  \begin{frame}
    \frametitle{Mercado}
    Antecedentes:
    \begin{itemize}
      \item 65\% de los chilenos está interesado en domótica.
      \item De estos, 47\% manifiesta interés en abordar problemas de salud y bienestar.
      de construcción.
      \item 2.1 millones de chilenos como potenciales consumidores.
    \end{itemize}

    Segmentación:
    \begin{itemize}
      \item Venta directa
      \item Resellers
    \end{itemize}
  \end{frame}


  \subsection{Análisis de competencia}
  \begin{frame}
    \frametitle{Análisis de competencia}
    Tres posibles competidores:
    \begin{itemize}
      \item \emph{SIDCO}: Empresa chilena, 9 años de experiencia. Su visión apunta a generar soluciones eficientes energeticamente orientadas al despliegue masivo para proyectos de larga envergadura.

      \item \emph{TAMED}: Compañía chilena dedicada a la implementación de soluciones en 
      automatización para inmobiliarias y constructoras. Resellers de la tecnología \emph{FIBARO} en Chile. 

      \item \emph{OZOM}: Autodefinido como ``Una nueva tecnología de productos que permiten 
      ser controlados a través de una aplicación''. Cuenta como reseller en Chile a Sodimac 
      HomeCenter, Apuntan principalmente al sistema de sensado e interacción directa con objetos electrónicos.
    \end{itemize}
  \end{frame}

  \subsection{Proveedores y proveedores clave}
  \begin{frame}
    \frametitle{Proveedores clave y riesgos}
    Proveedores clave
    \begin{itemize}
      \item Proveedores de productos plásticos
      \item Proveedores de componentes para electrónica y telecomunicación
    \end{itemize}

    Riesgos asociados
    \begin{itemize}
      \item Las ventas podrían no subir debido a una falla en la evaluación del plan 
      de marketing.
      \item Problemas con la adquisición de materiales y herramientas 
      desde el extranjero.
      \item Una falla en el diseño podría no ser atractivo al cliente lo cual 
      mermaría nuestras ventas.
    \end{itemize}
  \end{frame}
